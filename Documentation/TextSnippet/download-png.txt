[[DO GET:png withParams:nil withProgress:^(RXDownloadOperation* op, DODownProgress p){
        
        NSLog(@"Down progress = %f",(float)p.totalBytesWritten/p.totalBytesExpectedToWrite);
        
    } withCompletion:^(RXDownloadOperation* operation, NSError* error){
        
        if (operation.state == RXBOSuccessFinished)
        {
            if (!error) {
                NSData *data = [NSData dataWithContentsOfURL:operation.locationFile];
                if (!data)  {  NSLog(@"writeTmpFileOnDiskFromOperation data==nil");   return; }
                
                NSString* nameOfLastItemInPath  = [[Helper divorcePathOnArray:operation.url bySymbol:@"/"] lastObject];
                NSString  *pathToDocuments      = [NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES) firstObject];
                NSString* pathToWritingFile     = [pathToDocuments stringByAppendingPathComponent:nameOfLastItemInPath];
                NSError* errorWriting;
                BOOL isSuccess = [FCFileManager writeFileAtPath:pathToWritingFile content:data error:&errorWriting];
                
                if (errorWriting) { NSLog(@"(%@)writeTmpFileOnDiskFromOperation| errorWriting = %@",operation.name, errorWriting); return; }
                NSString* resultOfWriting = [NSString stringWithFormat:@"(%@) isSuccess writing = %@ | By path = %@", operation.name,
                                             (isSuccess) ? @"YES" : @"NO",pathToWritingFile];
                NSLog(@"%@",resultOfWriting);
       
            }
        }
  }] start];
