  NSLog(@“Progress = %f”, (float)p.countOfBytesReceived/p.countOfBytesExpectedToReceive);        
} withCompletion:^(RXDataTaskOperation *op, NSError *error) {
        
        NSError* errorConvertJSON;

							       error:&errorConvertJSON];
